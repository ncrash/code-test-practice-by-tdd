rem sonar build command with UTF-8 encoding
set MAVEN_OPTS= -Dfile.encoding=UTF-8
mvn clean install sonar:sonar 

rem How do I skip the tests?
rem  Add the parameter -Dmaven.test.skip=true or -DskipTests=true in the command line, depending on whether you want to skip test compilation and execution or only execution. See the example Skipping Tests in the Surefire Plugin's documentation for more details. 
set MAVEN_OPTS= -Dmaven.test.skip=true

rem Sometimes Maven will fail when running the test cases with this error.
rem https://confluence.slac.stanford.edu/pages/viewpage.action?pageId=38246
rem  java.lang.OutOfMemory
set MAVEN_OPTS=-Xmx1024m -XX:MaxPermSize=128m
